# Core Streamlit scanner runtime.
# Upper bounds cap major versions to prevent a surprise breaking release from
# breaking a deploy. For full reproducibility, replace with exact == pins from
# `pip freeze` in the deploy environment.
# 1.54.0: pip-audit flagged PYSEC-2026-212/-2285 in 1.49.1, and 1.49.1's
# pillow<12 bound blocked the Pillow CVE fixes below — one coupled upgrade.
streamlit==1.54.0
# Pillow is transitive (matplotlib/streamlit image handling) but pip-audit
# flagged 11 CVEs in 11.3.0; floor it at the fixed release.
pillow>=12.3,<14
pandas>=2.0,<3
numpy>=1.23,<3
yfinance>=0.2.27,<0.3
pytz>=2025.2
requests>=2.31,<3

# Universe loading / HTML table parsing.
# No upper cap: lxml is a C-extension and we need whatever major version ships
# a prebuilt wheel for the deploy's Python (3.14). Capping <6 forced 5.4.0,
# which has no cp314 wheel and fails to build from source on Streamlit Cloud.
lxml>=5.0
html5lib>=1.1,<2
beautifulsoup4>=4.12,<5

# Charts and scanner UI.
# altair<5.5: 5.5+ generates TypedDict(closed=True) schemas that crash on
# Python 3.14 (Streamlit Cloud runtime). Pin below that until upstream fixes it.
altair>=5.0,<5.5
plotly>=5.18,<7
# Circle-packing layout for the bubbles view of the score map (pure Python).
circlify>=0.15,<1
streamlit-authenticator>=0.4.0,<0.5
bcrypt>=4.0,<5
streamlit-autorefresh<2
streamlit-cookies-manager>=0.2.0,<0.3

# Scheduler, persistence, and Neon/Postgres access.
SQLAlchemy>=2.0,<3
psycopg[binary]>=3.2,<4

# Optional error monitoring (no-op unless SENTRY_DSN is set).
sentry-sdk>=2.0,<3
